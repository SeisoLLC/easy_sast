125@validate 

126def submit_artifacts(*, upload_api: UploadAPI) -> bool: 

127 """ 

128 Submit build artifacts to Veracode for SAST 

129 """ 

130 artifacts: List[Path] = [] 

131 

132 LOG.info("Attempting to create a build") 

133 if create_build(upload_api=upload_api): 

134 LOG.info("Successfully called create_build") 

135 else: 

136 LOG.error("Failed to call create_build") 

137 return False 

138 

139 LOG.info("Beginning pre-upload file filtering") 

140 for artifact in upload_api.build_dir.iterdir(): 

141 LOG.debug("Calling filter_file on %s", artifact) 

142 if filter_file(artifact=artifact): 

143 artifacts += [artifact] 

144 

145 # Check to see if the artifacts list is empty 

146 if not artifacts: 

147 LOG.error("Nothing to upload") 

148 return False 

149 

150 LOG.info("Beginning file uploads") 

151 for artifact in artifacts: 

152 if upload_large_file(upload_api=upload_api, artifact=artifact): 

153 LOG.info("Successfully uploaded %s", artifact) 

154 else: 

155 LOG.error("Failed to upload %s", artifact) 

156 return False 

157 

158 LOG.info("File uploads complete") 

159 if begin_prescan(upload_api=upload_api): 

160 LOG.info("Successfully began the prescan") 

161 else: 

162 LOG.error("Failed to start the prescan") 

163 return False 

164 

165 return True
