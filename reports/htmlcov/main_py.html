18def main() -> None:

20 Integration with Veracode Static Analysis

28 # Get the effective config

30 config = get_config() 

31 except ValueError: 

32 log.error("Unable to create a valid configuration")

35 # Update the log level to whatever was set in the config 

36 logging.getLogger().setLevel(config["loglevel"])

41 api=ResultsAPI(app_id=config["apis"]["results"]["app_id"]), config=config

43 upload_api = apply_config( 

44 api=UploadAPI(app_id=config["apis"]["upload"]["app_id"]), config=config

53 configure_environment(

55 )

59 log.error("Failed to submit build artifacts for scanning") 

60 sys.exit(1) 

61 elif step == "check_compliance": 

62 configure_environment( 

63 api_key_id=config["api_key_id"], api_key_secret=config["api_key_secret"]

66 sys.exit(1)

68 

69if __name__ == "__main__":
